~~~js
//普通函数 写完函数 + 调用
function fn(){
    console.log(1);
}
fn();

//1.立即执行函数：不需要调用 马上执行
//2.写法：
// (function(){})()  /  (function(){}());

//第二个小括号 相当于 调用的作用
(function(){
    console.log(1)
})();  
//if有多个立即执行函数 要用分号隔开

//第二个小括号 也可以传参
(function(a){
    console.log(a);
})(1)

//也可以非匿名函数
(function sum(a, b){
    console.log(a + b);
})(1 , 2)

//第二个写法与第一个相同

//3.最大的作用是 独立创建了一个作用域 避免了命名冲突问题
~~~
